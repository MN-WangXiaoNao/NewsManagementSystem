-- 创建实训的数据库
create database shixun DEFAULT CHARACTER SET UTF8;

use shixun;

-- 创建 Catalog 表
CREATE table catalog(
id INT PRIMARY KEY AUTO_INCREMENT,
name VARCHAR(20),
description VARCHAR(40)
);

show tables;

select * from catalog;

-- 创建 News 表
CREATE table news(
id INT PRIMARY KEY AUTO_INCREMENT,
title VARCHAR(20),
content VARCHAR(200),
author varchar(10),
publishedtime datetime,
source VARCHAR(20),
catalogId int
);
